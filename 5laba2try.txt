		sub ecx, 2

			mov eax, ecx
			cmp eax, 0

			//imul eax, 4
			shl eax, 2 // y = j*width*4
			add eax, edi // eax = buffer + y
			add eax, 1 // eax = buffer + y + 1

			jmp i_firstpass
			for_i :
		inc edx

			cmp edx, esi //i>?<width
